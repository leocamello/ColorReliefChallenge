Blending the Maps Together¶

In [6]:
input_ds = gdal.Open(input_dem)

geotransform = input_ds.GetGeoTransform()

origin_x = geotransform[0]
origin_y = geotransform[3]
pixel_width = geotransform[1]
pixel_height = geotransform[5]
columns = input_ds.RasterXSize
rows = input_ds.RasterYSize
bands = 3

output_dem = ".\Data\jotunheimen\jotunheimen_output.tif"

driver = gdal.GetDriverByName("GTiff")

output_ds = driver.Create(output_dem, columns, rows, bands, gdal.GDT_Byte)
output_ds.SetGeoTransform((origin_x, pixel_width, 0, origin_y, 0, pixel_height))

spatial_reference = osr.SpatialReference()
spatial_reference.ImportFromWkt(input_ds.GetProjectionRef())
output_ds.SetProjection(spatial_reference.ExportToWkt())

for band in range(bands):
    output_array = []
    for row in xrange(rows):
        output_array.append([])
        for column in xrange(columns):
            color_relief_background = color_relief_array[row][column][band]
            hillshade_layer = hillshade_array[row][column]
            slope_layer = slope_array[row][column]
            color = 0.1 * slope_layer + 0.4 * hillshade_layer
            color += 0.5 * color_relief_background
            output_array[row].append(color)

    output_array = np.array(output_array, dtype=np.uint8)

    output_band = output_ds.GetRasterBand(band+1)
    output_band.WriteArray(output_array)
    output_band.FlushCache()

input_ds = None
output_ds = None
    
plt.imshow(mpimg.imread(output_dem))
